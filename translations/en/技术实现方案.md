## 2. GitHub App Integration Plan

### 2.1 GitHub App Configuration

#### 2.1.1 Required Permissions

Based on MVP requirements, the GitHub App needs the following permissions:

| Permission Name | Permission Level | Purpose |
|---------|----------|---------|
| contents | write | Read/write repository content, create translation files and PRs |
| pull_requests | write | Create and manage Pull Requests |
| metadata | read | Read repository metadata, get repo info |

#### 2.1.2 GitHub App Creation Steps

1. Log in to GitHub Developer Settings  
2. Create new GitHub App  
3. Configure basic info:
   - App name: `GitHub Global`
   - Homepage URL: `https://your-domain.com`
   - Callback URL: `https://your-domain.com/api/auth/github/callback`
   - Webhook URL: `https://your-domain.com/api/webhooks/github`
4. Configure permissions: `contents: write`, `pull_requests: write`, `metadata: read`
5. Generate Webhook Secret  
6. Download Private Key (PEM format)  
7. Record App ID  

### 2.2 Octokit SDK Integration

#### 2.2.1 Initialize Octokit App

```typescript
import { App } from 'octokit';

export const app = new App({
```

### 3.3 Translation Request Wrapper

```typescript
interface TranslationRequest {
  sourceLang: string;
  targetLang: string;
  content: string;
  model?: string;
}

interface TranslationResponse {
  translatedContent: string;
  model: string;
  tokensUsed: number;
}

export async function translateText(
  request: TranslationRequest
): Promise<TranslationResponse> {
  const model = request.model || 'deepseek/deepseek-chat';
  
  const prompt = buildTranslationPrompt(
    request.sourceLang,
    request.targetLang,
    request.content
  );

  const completion = await openai.chat.completions.create({
    model,
    messages: [
      {
        role: 'user',
        content: prompt,
      },
    ],
    temperature: 0.3,
    max_tokens: 4000,
  });

  return {
    translatedContent: completion.choices[0].message.content || '',
    model: completion.model,
    tokensUsed: completion.usage?.total_tokens || 0,
  };
}

function buildTranslationPrompt(
  sourceLang: string,
  targetLang: string,
  content: string
): string {
  return `You are a professional technical document translator. 
Translate the following Markdown content from ${sourceLang} to ${targetLang}.

Requirements:
1. Preserve all Markdown formatting (headers, code blocks, links, tables, etc.)
2. Keep technical terms unchanged when appropriate
3. Maintain original document structure
4. Only translate the content, do not add or remove information
5. Ensure translation is natural and professional
6. Do not translate code blocks unless explicitly requested

Source content:
${content}

Translation:`;
}
```

### 3.4 Model Selection Strategy

```typescript
interface ModelConfig {
  id: string;
  name: string;
  provider: string;

const SUPPORTED_MODELS: ModelConfig[] = [
  {
    id: 'deepseek/deepseek-chat',

  },

  },
  {
    id: 'zhipu/glm-4',
    name: 'Zhipu AI',
    provider: 'Zhipu AI',
    languages: ['zh-CN', 'en'],
    priority: 1,
  },
  {
    id: 'openai/gpt-4',
    name: 'GPT-4',
    provider: 'OpenAI',
    languages: ['en', 'zh-CN', 'ja', 'ko'],
    priority: 1,
  },
  {
    id: 'anthropic/claude-3-opus',
    name: 'Claude 3',
    provider: 'Anthropic',
    languages: ['en', 'zh-CN'],
    priority: 2,
  },
  {
    id: 'google/gemini-pro',
    name: 'Gemini Pro',
    provider: 'Google',
    languages: ['en', 'zh-CN', 'ja', 'ko'],
    priority: 2,
  },
];

export function selectBestModel(
  sourceLang: string,
  targetLang: string
): string {
  const eligibleModels = SUPPORTED_MODELS.filter(
    (model) =>
      model.languages.includes(sourceLang) &&
      model.languages.includes(targetLang)
  );

  eligibleModels.sort((a, b) => a.priority - b.priority);

  return eligibleModels[0]?.id || 'deepseek/deepseek-chat';
}
```
