## 2. GitHub App 集成方案

### 2.1 GitHub App 配置

#### 2.1.1 必需权限

根据 MVP 需求,GitHub App 需要以下权限:

| 权限名称 | 权限级别 | 用途说明 |
|---------|----------|---------|
| contents | write | 读写仓库内容,创建翻译文件和 PR |
| pull_requests | write | 创建和管理 Pull Request |
| metadata | read | 读取仓库元数据,获取仓库信息 |

#### 2.1.2 GitHub App 创建步骤

1. 登录 GitHub Developer Settings
2. 创建新的 GitHub App
3. 配置基本信息:
   - App 名称: `GitHub Global`
   - Homepage URL: `https://your-domain.com`
   - Callback URL: `https://your-domain.com/api/auth/github/callback`
   - Webhook URL: `https://your-domain.com/api/webhooks/github`
4. 配置权限: `contents: write`, `pull_requests: write`, `metadata: read`
5. 生成 Webhook Secret
6. 下载 Private Key (PEM 格式)
7. 记录 App ID

### 3.3 翻译请求封装

```typescript
interface TranslationRequest {
  sourceLang: string;
  targetLang: string;
  content: string;
  model?: string;
}

interface TranslationResponse {
  translatedContent: string;
  model: string;
  tokensUsed: number;
}

export async function translateText(
  request: TranslationRequest
): Promise<TranslationResponse> {
  const model = request.model || 'deepseek/deepseek-chat';
  
  const prompt = buildTranslationPrompt(
    request.sourceLang,
    request.targetLang,
    request.content
  );

  const completion = await openai.chat.completions.create({
    model,
    messages: [
      {
        role: 'user',
        content: prompt,
      },
    ],
    temperature: 0.3,
    max_tokens: 4000,
  });

  return {
    translatedContent: completion.choices[0].message.content || '',
    model: completion.model,
    tokensUsed: completion.usage?.total_tokens || 0,
  };
}

function buildTranslationPrompt(
  sourceLang: string,
  targetLang: string,
  content: string
): string {
  return `You are a professional technical document translator. 
Translate the following Markdown content from ${sourceLang} to ${targetLang}.

Requirements:
1. Preserve all Markdown formatting (headers, code blocks, links, tables, etc.)
2. Keep technical terms unchanged when appropriate
3. Maintain original document structure
4. Only translate the content, do not add or remove information
5. Ensure translation is natural and professional
6. Do not translate code blocks unless explicitly requested

Source content:
${content}

Translation:`;
}
```

```typescript
export function selectBestModel(
  sourceLang: string,
  targetLang: string
): string {
  const eligibleModels = SUPPORTED_MODELS.filter(
    (model) =>
      model.languages.includes(sourceLang) &&
      model.languages.includes(targetLang)
  );

  eligibleModels.sort((a, b) => a.priority - b.priority);

  return eligibleModels[0]?.id || 'deepseek/deepseek-chat';
}
```

#### 5.1.1 OAuth 授权流程

```typescript
import { app } from '@/lib/github';
import { prisma } from '@/lib/prisma';

export async function handleGitHubAuth(code: string, state: string) {
  try {
    const { authentication } = await app.oauth.createOAuthUserAuthentication({
      code,
    });

    const githubUser = await fetchGitHubUser(authentication.token);

    let user = await prisma.user.findUnique({
      where: { githubId: BigInt(githubUser.id) },
    });

    if (!user) {
      user = await prisma.user.create({
        data: {
          githubId: BigInt(githubUser.id),
          githubLogin: githubUser.login,
          githubAvatarUrl: githubUser.avatar_url,
          accessToken: encryptToken(authentication.token),
          freeQuota: 1000000,
          usedQuota: 0,
        },
      });
    } else {
      user = await prisma.user.update({
        where: { id: user.id },
        data: {
          accessToken: encryptToken(authentication.token),
          githubAvatarUrl: githubUser.avatar_url,
        },
      });
    }

    const jwtToken = generateJWT(user);

    return { user, token: jwtToken };
  } catch (error) {
    console.error('GitHub auth error:', error);
    throw new Error('Authentication failed');
  }
}

async function fetchGitHubUser(token: string) {
  const octokit = new Octokit({ auth: token });
  const { data } = await octokit.rest.users.getAuthenticated();
  return data;
}
```

```typescript
export async function scanRepositoryFiles(
  repositoryId: bigint,
  octokit: any
) {
  const { data: files } = await octokit.rest.repos.getContent({
    owner: 'owner',
    repo: 'repo',
    path: '',
  });

  const mdFiles = scanMarkdownFiles(files);
  return mdFiles;
}

function scanMarkdownFiles(files: any[]): string[] {
  const mdFiles: string[] = [];

  function traverse(items: any[]) {
    for (const item of items) {
      if (item.type === 'file' && item.name.endsWith('.md')) {
        mdFiles.push(item.path);
      } else if (item.type === 'dir') {
        if (shouldIncludeDir(item.name)) {
          traverse(item.entries || []);
        }
      }
    }
  }

  traverse(files);
  return mdFiles;
}

function shouldIncludeDir(dirName: string): boolean {
  const excludeDirs = ['.git', 'node_modules', '.next', 'dist'];
  return !excludeDirs.includes(dirName);
}
```

### 5.3 翻译引擎模块

#### 5.3.1 批量翻译

```typescript
import { translateText, selectBestModel } from '@/lib/openrouter';
import { prisma } from '@/lib/prisma';
import { calculateFileHash } from '@/lib/hash';

export async function batchTranslate(
  repositoryId: bigint,
  files: string[],
  targetLanguages: string[]
) {
  const octokit = await getInstallationOctokit(repositoryId);
  const repository = await prisma.repository.findUnique({
    where: { id: repositoryId },
  });

  const task = await prisma.translationTask.create({
    data: {
      repositoryId,
      triggerType: 'manual',
      totalFiles: BigInt(files.length * targetLanguages.length),
      status: 'processing',
      startedAt: new Date(),
    },
  });

  let completedCount = 0;
  let failedCount = 0;

  for (const lang of targetLanguages) {
    for (const file of files) {
      try {
        const { data: content } = await octokit.rest.repos.getContent({
          owner: repository.fullName.split('/')[0],
          repo: repository.fullName.split('/')[1],
          path: file,
          mediaType: 'raw',
        });

        const fileHash = calculateFileHash(content);
        
        const existingTranslation = await prisma.translation.findUnique({
          where: {
            repositoryId_filePath_targetLanguage: {
              repositoryId,
              filePath: file,
              targetLanguage: lang,
            },
          },
        });

        if (existingTranslation && existingTranslation.fileHash === fileHash) {
          completedCount++;
          continue;
        }

        const model = selectBestModel(repository.defaultLanguage, lang);
        const { translatedContent, tokensUsed } = await translateText({
          sourceLang: repository.defaultLanguage,
          targetLang: lang,
          content: content.toString(),
          model,
        });

        const targetPath = `translations/${lang}/${file}`;
        await octokit.rest.repos.createOrUpdateFileContents({
          owner: repository.fullName.split('/')[0],
          repo: repository.fullName.split('/')[1],
          path: targetPath,
          message: `Translate ${file} to ${lang}`,
          content: Buffer.from(translatedContent).toString('base64'),
          branch: `translation/${lang}`,
        });

        await prisma.translation.create({
          data: {
            repositoryId,
            filePath: file,
            targetLanguage: lang,
            sourceLanguage: repository.defaultLanguage,
            fileHash,
            fileSize: BigInt(content.toString().length),
            status: 'completed',
            translatedAt: new Date(),
          },
        });

        await updateUserQuota(repository.userId, tokensUsed);
        completedCount++;
      } catch (error) {
        console.error(`Translation failed for ${file}:`, error);
        
        await prisma.translation.create({
          data: {
            repositoryId,
            filePath: file,
            targetLanguage: lang,
            sourceLanguage: repository.defaultLanguage,
            status: 'failed',
            errorMessage: error.message,
          },
        });

        failedCount++;
      }
    }
  }

  await prisma.translationTask.update({
    where: { id: task.id },
    data: {
      completedFiles: BigInt(completedCount),
      failedFiles: BigInt(failedCount),
      status: 'completed',
      completedAt: new Date(),
    },
  });

  await createPullRequest(repositoryId, octokit, targetLanguages);
}
```

```typescript
import crypto from 'crypto';
import { app } from '@/lib/github';
import { prisma } from '@/lib/prisma';

export async function handleWebhook(
  payload: any,
  signature: string
): Promise<void> {
  const payloadString = JSON.stringify(payload);

  if (!verifyWebhookSignature(payloadString, signature)) {
    throw new Error('Invalid webhook signature');
  }

  const webhookLog = await prisma.webhookLog.create({
    data: {
      eventType: payload.event_type,
      payload: payloadString,
    },
  });

  if (payload.event_type === 'push') {
    await handlePushEvent(payload);
  }

  await prisma.webhookLog.update({
    where: { id: webhookLog.id },
    data: { processed: true },
  });
}

function verifyWebhookSignature(
  payload: string,
  signature: string
): boolean {
  const hmac = crypto.createHmac('sha256', process.env.GITHUB_WEBHOOK_SECRET!);
  const digest = hmac.update(payload).digest('hex');
  return `sha256=${digest}` === signature;
}

async function handlePushEvent(payload: any) {
  const { repository, commits } = payload;
  
  const dbRepository = await prisma.repository.findUnique({
    where: {
      githubRepoId: BigInt(repository.id),
    },
  });

  if (!dbRepository || !dbRepository.enabled) {
    return;
  }

  const changedFiles = await getChangedFiles(
    dbRepository.lastCommitSha,
    commits[commits.length - 1].id
  );

  const mdFiles = changedFiles.filter((file) => file.endsWith('.md'));

  if (mdFiles.length === 0) {
    return;
  }

  await batchTranslate(dbRepository.id, mdFiles, dbRepository.targetLanguages);

  await prisma.repository.update({
    where: { id: dbRepository.id },
    data: {
      lastSyncedAt: new Date(),
      lastCommitSha: commits[commits.length - 1].id,
    },
  });
}
```

```typescript
export async function insertLanguageLinks(
  repositoryId: bigint,
  octokit: any
): Promise<void> {
  const repository = await prisma.repository.findUnique({
    where: { id: repositoryId },
  });

  const { data: readme } = await octokit.rest.repos.getContent({
    owner: repository.fullName.split('/')[0],
    repo: repository.fullName.split('/')[1],
    path: 'README.md',
    mediaType: 'raw',
  });

  const content = readme.toString();
  const languageLinks = generateLanguageLinks(repository.targetLanguages);

  const updatedContent = insertLanguageLinks(content, languageLinks);

  await octokit.rest.repos.createOrUpdateFileContents({
    owner: repository.fullName.split('/')[0],
    repo: repository.fullName.split('/')[1],
    path: 'README.md',
    message: 'Add language links',
    content: Buffer.from(updatedContent).toString('base64'),
    branch: `translation/${repository.targetLanguages[0]}`,
  });
}

function generateLanguageLinks(languages: string[]): string {
  const languageNames: Record<string, string> = {
    'en': 'English',
    'zh-CN': '中文',
    'ja': '日本語',
    'ko': '한국어',
    'es': 'Español',
    'fr': 'Français',
    'de': 'Deutsch',
  };

  const links = languages
    .map((lang) => `- [${languageNames[lang] || lang}](../${lang}/README.md)`)
    .join('\n');

  return `## Languages\n\n${links}`;
}

function insertLanguageLinks(content: string, links: string): string {
  const languagesHeader = '## Languages';
  const languagesHeaderIndex = content.indexOf(languagesHeader);

  if (languagesHeaderIndex !== -1) {
    return content + '\n\n' + links;
  }

  const nextNewline = content.indexOf('\n', languagesHeaderIndex);
  const insertPosition = nextNewline !== -1 
    ? nextNewline + 1 
    : languagesHeaderIndex + languagesHeader.length;

  return (
    content.slice(0, insertPosition) + 
    '\n' + 
    links + 
    content.slice(insertPosition)
  );
}
```

#### 5.6.1 API Key 加密

```typescript
import crypto from 'crypto';

const ALGORITHM = 'aes-256-cbc';
const KEY_LENGTH = 32;
const IV_LENGTH = 16;

export function encryptApiKey(apiKey: string): string {
  const key = Buffer.from(process.env.ENCRYPTION_KEY!, 'hex').slice(0, KEY_LENGTH);
  const iv = crypto.randomBytes(IV_LENGTH);
  const cipher = crypto.createCipheriv(ALGORITHM, key, iv);
  let encrypted = cipher.update(apiKey, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  return iv.toString('hex') + ':' + encrypted;
}

export function decryptApiKey(encryptedData: string): string {
  const parts = encryptedData.split(':');
  const iv = Buffer.from(parts[0], 'hex');
  const encrypted = parts[1];
  const key = Buffer.from(process.env.ENCRYPTION_KEY!, 'hex').slice(0, KEY_LENGTH);
  const decipher = crypto.createDecipheriv(ALGORITHM, key, iv);
  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  return decrypted;
}
```

```typescript
interface QuotaCheck {
  allowed: boolean;
  remaining: bigint;
  resetAt?: Date;
}

export async function checkUserQuota(userId: bigint): Promise<QuotaCheck> {
  const user = await prisma.user.findUnique({
    where: { id: userId },
  });

  if (!user) {
    return { allowed: false, remaining: 0n };
  }

  const now = new Date();
  const resetDate = user.quotaResetAt || new Date(now.getFullYear(), now.getMonth(), now.getDate());

  if (now > resetDate) {
    await prisma.user.update({
      where: { id: userId },
      data: {
        usedQuota: 0n,
        quotaResetAt: now,
      },
    });

    return { allowed: true, remaining: user.freeQuota };
  }

  const remaining = user.freeQuota - user.usedQuota;
  return { allowed: remaining > 0, remaining };
}

export async function updateUserQuota(
  userId: bigint,
  tokensUsed: number
): Promise<void> {
  await prisma.user.update({
    where: { id: userId },
    data: {
      usedQuota: {
        increment: BigInt(tokensUsed),
      },
    },
  });
}
```

```typescript
export default function TranslationProgressPage({ params }: { params: { taskId: string } }) {

      setLoading(false);

    ? Math.round((progress.completedFiles / progress.totalFiles) * 100)
```

### 14.3 常见问题

**Q: 如何切换 AI 模型?**

A: 在仓库配置页面选择目标语言时,系统会自动根据源语言和目标语言选择最优模型。也可以在 `translationConfig` 中手动指定模型。

**Q: 如何避免重复翻译?**

A: 系统通过文件 Hash 值检测变更,只有当文件内容发生变化时才会重新翻译。Hash 值存储在 `translations` 表中。

**Q: 如何处理翻译失败?**

A: 翻译失败的文件会被标记为 `failed` 状态,并记录错误信息。用户可以在翻译进度页面查看失败文件,并手动重试。

**Q: 免费额度如何计算?**

A: 每个账号每天 100 万字符免费额度。额度在每天 0 点重置。超出额度后需要等待第二天或升级套餐。
